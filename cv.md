### Code Example

```
function BinarySearch(sortArr, val) {
    let start = 0;
    let end = sortArr.length - 1;

    while (start <= end) {
        let mid = Math.trunc((start + end) / 2);

        if (sortArr[mid] === val) return mid;
        if (sortArr[mid] < val) start = mid + 1;
        if (sortArr[mid] > val) end = mid - 1;
    }

    return -1;
}
```
